    "def extract_text_from_pdf(pdf_path):\n",
    "    \"\"\"Extract text from PDF file\"\"\"\n",
    "    try:\n",
    "        import pymupdf  # or PyPDF2\n",
    "        doc = pymupdf.open(pdf_path)\n",
    "        text = \"\"\n",
    "        for page in doc:\n",
    "            text += page.get_text()\n",
    "        return text\n",
    "    except Exception as e:\n",
    "        print(f\"Error reading {pdf_path}: {e}\")\n",
    "        return None\n",
    "\n",
    "def extract_text_from_txt(txt_path):\n",
    "    \"\"\"Extract text from TXT file\"\"\"\n",
    "    try:\n",
    "        with open(txt_path, 'r', encoding='utf-8') as f:\n",
    "            return f.read()\n",
    "    except Exception as e:\n",
    "        print(f\"Error reading {txt_path}: {e}\")\n",
    "        return None\n",

    "    else:\n",
    "        continue\n",
    "    \n",
    "    if text:\n",
    "        paper_name = path.split('/')[-1].replace('.pdf', '').replace('.txt', '')\n",

  },
  {
   "cell_type": "code",
   "execution_count": None,
   "metadata": {},
   "outputs": [],
   "source": [
    "# Clean the text data\n",
    "def clean_text(text):\n",
    "    \"\"\"Clean text for NLP analysis\"\"\"\n",
    "    # Lowercase\n",
    "    text = text.lower()\n",
    "    \n",
    "    # Remove punctuation\n",
    "    text = re.sub(r'[^\\w\\s]', '', text)\n",
    "    \n",
    "    # Remove numbers\n",
    "    text = re.sub(r'\\d+', '', text)\n",
    "    \n",
    "    # Remove URLs\n",
    "    text = re.sub(r'http\\S+|\\S+\\.com\\S+', '', text)\n",
    "    \n",
    "    # Remove citations (e.g., [12], (2023))\n",
    "    text = re.sub(r'\\[\\d+\\]', '', text)\n",
    "    text = re.sub(r'\\(\\d{4}\\)', '', text)\n",
    "    \n",
    "    # Normalize whitespace\n",
    "    text = re.sub(r'\\s+', ' ', text).strip()\n",
    "    \n",
    "    return text\n",
    "\n",
    "# Clean all papers\n",
    "paper_texts_cleaned = {}\n",
    "for paper_name, text in paper_texts.items():\n",
    "    paper_texts_cleaned[paper_name] = clean_text(text)\n",
    "\n",
    "print(f\"Cleaned {len(paper_texts_cleaned)} papers\")\n",
    "print(f\"\\nExample cleaned text (first 200 chars):\")\n",
    "print(paper_texts_cleaned[list(paper_texts_cleaned.keys())[0]][:200])"
   ]
